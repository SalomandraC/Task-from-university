#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main()
{
    //игициализация переменных
    int a, s, b, t, k, n;
    /*
    a-час отправления
    b-минута отправления
    s-минут в пути
    t-часов в пути
    k-ответ, часов
    n-ответ, минут
    */
    //инициализация дополнительной символьной переменной
    char c;
    c=':';
    //ввод
    scanf("%d:%d", &a, &b);
    scanf("%d %d", &t, &s);
     
    //формулы для нахождения часов с рассмотрением частного случая,
    //когда асы в суммее дают 23, а минуты 1 из за чего (a+t)%24 поздно срабатывает
    k= ((a+t)%24+(b+s)/60 == 24) ? 0 : ((a+t)%24+(b+s)/60);
    n=(b+s)%60;
     
    //вывод
    printf("%02d%c%02d", k, c, n);
  
    return 0;
}
