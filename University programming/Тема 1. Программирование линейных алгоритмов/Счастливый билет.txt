#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
 
using namespace std;
 
int main() {
    //инициализация переменных
    int n, n1, n2, s, p;
    /*
    n-вводимое число
    n1-последние 3 цифры числа
    n2-первые 3 цифры числа
    s-сумма последних 3х цифр
    p-сумма первых 3х цифр
    */
     
    //ввод
    cin >> n;
 
    n1 = n % 1000;
    n2 = n / 1000;
    //нахождение суммы последних 3х цифр
    s += n1 % 10; n1 = n1 / 10;
    s += n1 % 10; n1 = n1 / 10;
    s += n1 % 10;
    //нахождение суммы первых 3х цифр
    p += n2 % 10; n2 = n2 / 10;
    p += n2 % 10; n2 = n2 / 10;
    p += n2 % 10;
     
    //вывод
    (s == p) ? (cout << "YES") : (cout << "NO");
 
    return 0;
}
